    <script>
        function preview(event, id) {
            const reader = new FileReader();
            reader.onload = function(){
                const out = document.getElementById(id);
                out.src = reader.result;
                out.style.display = 'block';
                out.nextElementSibling.nextElementSibling.nextElementSibling.style.display = 'none';
            };
            reader.readAsDataURL(event.target.files[0]);
        }

        const btn = document.getElementById('scrollBtn');
        window.addEventListener('scroll', () => {
            const y = window.scrollY;
            const h = window.innerHeight;
            if (y < h * 0.5) { btn.innerText = "PAGE 2 →"; btn.href = "#slide2"; }
            else if (y < h * 1.5) { btn.innerText = "PAGE 3 →"; btn.href = "#slide3"; }
            else { btn.innerText = "← BACK TO PAGE 1"; btn.href = "#slide1"; }
        });
    </script>
